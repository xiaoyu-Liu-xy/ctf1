import struct
import binascii
import os

m = open("lxy.png","rb").read()
for i in range(1024):
    c = m[12:16] + struct.pack('>i', i) + m[20:29]
    crc = binascii.crc32(c) & 0xffffffff
    if crc == 0xc02f6b4f:
        print(i)
